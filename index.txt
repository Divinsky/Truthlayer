const PATTERNS = [
  // High-impact authority/citation claims
  { id: "statute", weight: 18, level: "High",
    re: /\b(section\s*\d+|§\s*\d+|article\s*\d+|chapter\s*\d+|title\s*\d+|penal code|civil code|act|statute|regulation|directive)\b/i,
    cat: "legal-citation",
    why: "Cites a specific statute or legal provision. Laws and section numbers change and are easy to misquote.",
    next: "Verify the section and current text on the official government source." },

  { id: "case-cite", weight: 22, level: "High",
    re: /\b[A-Z][a-zA-Z\.]+\s+v\.?\s+[A-Z][a-zA-Z\.]+/,
    cat: "legal-citation",
    why: "Quotes a specific court case. AI tools have a known pattern of inventing case citations.",
    next: "Look up the case on a primary legal database such as CourtListener or Westlaw before relying on it." },

  // Numbers + claims of fact
  { id: "specific-pct", weight: 9, level: "Medium",
    re: /\b\d{1,3}(\.\d+)?\s*%/,

    next: "Find the original study or dataset and confirm the figure and methodology." },

  { id: "money", weight: 8, level: "Medium",

    next: "Confirm the amount against a primary source dated within the last 12 months." },

  { id: "dosage", weight: 16, level: "High",

    cat: "medical",
    why: "Specifies a medical dose. Wrong dosing information can cause real harm.",
    next: "Confirm dosing with a licensed clinician, the drug label, or a trusted source like the NIH or NHS." },

  { id: "duration", weight: 5, level: "Low",
    re: /\b(within|in)\s+\d+\s+(hours?|days?|weeks?|months?|years?)\b/i,
    cat: "timeframe",
    why: "Asserts a specific timeframe or deadline.",
    next: "Verify the deadline in the official policy or law before acting on it." },

  { id: "year", weight: 4, level: "Low",
    re: /\b(19|20)\d{2}\b/,
    cat: "date",

    next: "Check the year against a primary source." },

  // Authority / publications
  { id: "study", weight: 12, level: "High",

/* Sentence split that keeps punctuation; protects decimals + abbreviations */
function splitSentences(text) {
  let t = text.trim().replace(/\s+/g, " ");
  // Protect decimals like 1.2 and 502(c) section refs
  t = t.replace(/(\d)\.(\d)/g, "$1․$2");
  // Protect common abbreviations from being treated as sentence ends
  t = t.replace(/\b(Mr|Mrs|Ms|Dr|Jr|Sr|St|vs|v|U\.S|e\.g|i\.e|etc|Inc|Ltd|Corp|No|Co|S&P)\./gi, "$1․");
  const parts = t.match(/[^.!?]+[.!?]+(\s|$)|[^.!?]+$/g) || [t];
  return parts.map(s => s.replace(/․/g, ".").trim()).filter(s => s.length > 4);
}

function analyzeSentence(s) {
  const matches = [];
  for (const p of PATTERNS) {
    if (p.re.test(s)) matches.push(p);
  }
  if (matches.length === 0) return null;

  const weight = matches.reduce((sum, m) => sum + m.weight, 0);
  // Determine level: any High pattern => High; otherwise sum-based
  const hasHigh = matches.some(m => m.level === "High");
  const hasMed = matches.some(m => m.level === "Medium");
  let level = "Low";
  if (hasHigh || weight >= 22) level = "High";
  else if (hasMed || weight >= 10) level = "Medium";

  // Compose plain-English why & verifyNext (deduped by category)
  const seenCat = new Set();
  const whys = [];
  const nexts = [];
  for (const m of matches) {
    if (seenCat.has(m.cat)) continue;
    seenCat.add(m.cat);
    whys.push(m.why);
    nexts.push(m.next);
  }
  return {
    claim: s,
    riskLevel: level,
    whyRisky: whys.slice(0, 2).join(" "),
    verifyNext: nexts[0],
    weight,
    cats: [...seenCat],
  };
}

function buildRewrite(text, flagged) {
  let out = text;

  // Soften absolutes and directives
  const softenings = [
    [/\bguaranteed?\b/gi, "likely (not guaranteed)"],
    [/\balways\b/gi, "often"],
    [/\bnever\b/gi, "rarely"],
    [/\bexactly\b/gi, "approximately"],
    [/\bcompletely?\b/gi, "largely"],
    [/\bdefinitely\b/gi, "likely"],
    [/\bcertainly\b/gi, "probably"],
    [/\ball customers\b/gi, "most customers"],
    [/\bin all cases\b/gi, "in many cases"],
    [/\bin all 50 states\b/gi, "in many states"],
    [/\bno exceptions\b/gi, "with limited exceptions"],
    [/\bno questions asked\b/gi, "subject to review"],
    [/\bsafe for most adults\b/gi, "described as generally tolerated, though individuals vary"],
    [/\bis safe\b/gi, "may be tolerated"],
    [/\bis the standard treatment\b/gi, "is one option clinicians may consider"],
    [/\byou must\b/gi, "you may be required to"],
    [/\byou are clearly liable\b/gi, "your liability is unclear and depends on the facts"],
    [/\byou should\b/gi, "you may want to"],
    [/\byou are entitled to\b/gi, "you may be entitled to"],
  ];
  for (const [re, replacement] of softenings) out = out.replace(re, replacement);

  // Add a gentle preamble + caveat
  const preamble = "Note: this version uses more careful language and removes claims that need verification.\n\n";
  const caveat = " — Specific figures, citations, statutes, and dates above should be verified against a primary source before you rely on them.";
  return preamble + out.trim() + caveat;
}

function buildChecklist(flagged) {
  const cats = new Set();
  flagged.forEach(c => c.cats.forEach(x => cats.add(x)));
  const items = [];
  if (cats.has("legal-citation")) items.push("Look up every statute, section number, and case citation on a primary legal source.");
  if (cats.has("study") || cats.has("statistic")) items.push("Confirm any cited study or statistic exists, by title, on the original publisher's site.");
  if (cats.has("medical") || cats.has("dosage")) items.push("Run any medical claim — especially dosing — past a licensed clinician.");
  if (cats.has("financial") || cats.has("money")) items.push("Sanity-check financial figures against a regulated source (SEC, broker disclosure, official statement).");
  if (cats.has("absolute") || cats.has("directive")) items.push("Replace absolute words (always, never, guaranteed) with the actual conditions and exceptions.");
  if (cats.has("authority") || cats.has("appeal")) items.push("For every “experts say” claim, ask: which expert, which document, which date?");
  if (cats.has("date") || cats.has("timeframe")) items.push("Confirm dates and deadlines against the official source they reference.");
  if (items.length < 3) {
    items.push("Read the text out loud — claims that sound suspiciously precise usually need a source.");
    items.push("Ask yourself: would I rely on this if it weren't written by an AI?");
  }
  return items.slice(0, 5);
}

function summarize(score, verdict, flagged) {
  if (flagged.length === 0) {
    return "No specific risky claims jumped out, but always sanity-check anything an AI tells you before relying on it.";
  }
  const cats = new Set();
  flagged.forEach(c => c.cats.forEach(x => cats.add(x)));
  const focus = [];
  if (cats.has("medical") || cats.has("dosage")) focus.push("medical");
  if (cats.has("legal-citation") || cats.has("case-cite")) focus.push("legal");
  if (cats.has("financial")) focus.push("financial");
  if (cats.has("study") || cats.has("statistic")) focus.push("statistical");
  const focusText = focus.length ? ` Mostly ${focus.slice(0,2).join(" and ")} claims.` : "";
  const word = verdict === "High Risk" ? "many" : verdict === "Medium Risk" ? "several" : "a few";
  return `${verdict.toLowerCase().replace(/^./, c => c.toUpperCase())} — ${word} statements need verification before you rely on this.${focusText}`;
}

function analyze(text) {
  const sentences = splitSentences(text);
  const scored = sentences.map(analyzeSentence).filter(Boolean);
  // Sort by weight desc, take top ~6
  scored.sort((a, b) => b.weight - a.weight);
  const flagged = scored.slice(0, 6);

  // Score = weighted sum with a soft cap so very heavy texts don't all peg at 100
  const total = scored.reduce((s, c) => s + c.weight, 0);
  // soft saturation: 0 → 0, 30 → 55, 60 → 75, 100 → 88, ∞ → ~95
  let score = Math.round(95 * (1 - Math.exp(-total / 45)));
  // density bonus: many sentences flagged out of total
  if (sentences.length > 0) {
    const density = scored.length / sentences.length;
    score = Math.min(98, Math.round(score + density * 10));
  }
  if (text.trim().length < 60) score = Math.min(score, 35);
  if (scored.length === 0) score = Math.max(8, Math.min(score, 20));

  let verdict = "Low Risk";
  if (score >= 65) verdict = "High Risk";
  else if (score >= 32) verdict = "Medium Risk";

  return {
    riskScore: score,
    verdict,
    summary: summarize(score, verdict, flagged),
    flaggedClaims: flagged.map(({ claim, riskLevel, whyRisky, verifyNext }) => ({
      claim, riskLevel, whyRisky, verifyNext
    })),
    saferVersion: buildRewrite(text, flagged),
    verifyChecklist: buildChecklist(flagged),
  };
}

const RING_C = 2 * Math.PI * 52; // ~326.7

function colorForScore(score) {
  if (score >= 65) return "var(--high)";
  if (score >= 32) return "var(--med)";
  return "var(--low)";
}
function levelClass(level) {
  return level === "High" ? "high" : level === "Medium" ? "med" : "low";
}
function verdictClass(v) {
  return v === "High Risk" ? "high" : v === "Medium Risk" ? "med" : "low";
}

function escapeHtml(s) {
  return String(s)
    .replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;").replace(/'/g, "&#39;");
}

function render(result) {
  // Gauge
  const score = Math.max(0, Math.min(100, result.riskScore));
  gaugeFill.setAttribute("stroke", colorForScore(score));
  // Animate from 0
  gaugeFill.setAttribute("stroke-dashoffset", RING_C);
  gaugeNum.textContent = "0";
  requestAnimationFrame(() => {
    gaugeFill.setAttribute("stroke-dashoffset", RING_C * (1 - score / 100));
    animateNumber(gaugeNum, 0, score, 800);
  });

  // Verdict badge
  verdictBadge.className = "verdict-badge " + verdictClass(result.verdict);
  verdictText.textContent = result.verdict;
  summary.textContent = result.summary;

  // Claims
  claimsCount.textContent = result.flaggedClaims.length
    ? `(${result.flaggedClaims.length})` : "";
  if (result.flaggedClaims.length === 0) {
    claimsList.innerHTML = `<div class="empty">No specific claims were flagged. Still, sanity-check anything important.</div>`;
  } else {
    claimsList.innerHTML = result.flaggedClaims.map(c => `
      <div class="claim">
        <div class="claim-head">
          <span class="risk-tag ${levelClass(c.riskLevel)}">${escapeHtml(c.riskLevel)} risk</span>
        </div>
        <div class="claim-text">&ldquo;${escapeHtml(c.claim)}&rdquo;</div>
        <div class="claim-meta">
          <div>
            <div class="label">Why it may be risky</div>
            <div class="value">${escapeHtml(c.whyRisky)}</div>
          </div>
          <div>
            <div class="label">What to verify next</div>
            <div class="value"><strong>${escapeHtml(c.verifyNext)}</strong></div>
          </div>
        </div>
      </div>
    `).join("");
  }

  // Safer version
  safer.textContent = result.saferVersion;

  // Checklist
  checklist.innerHTML = result.verifyChecklist.map(item => `
    <li>
      <span class="box">
        <svg width="11" height="11" viewBox="0 0 24 24" fill="none" stroke="#0d0a08" stroke-width="3.5" stroke-linecap="round" stroke-linejoin="round"><path d="M5 12l5 5L20 7"/></svg>
      </span>
      <span class="text">${escapeHtml(item)}</span>
    </li>
  `).join("");
  checklist.querySelectorAll("li").forEach(li => {
    li.addEventListener("click", () => li.classList.toggle("done"));
  });

  results.classList.add("is-visible");
  results.scrollIntoView({ behavior: "smooth", block: "start" });
}

function animateNumber(el, from, to, duration) {
  const start = performance.now();
  function tick(now) {
    const t = Math.min(1, (now - start) / duration);
    const eased = 1 - Math.pow(1 - t, 3);
    el.textContent = Math.round(from + (to - from) * eased);
    if (t < 1) requestAnimationFrame(tick);
  }
  requestAnimationFrame(tick);
}

checkBtn.addEventListener("click", () => {
  const text = input.value.trim();
  if (text.length < 12) return;

  // Loading state
  const labelEl = checkBtn.querySelector(".btn-label");
  const originalLabel = labelEl.textContent;
  const iconEl = checkBtn.querySelector("svg");
  iconEl.style.display = "none";
  const spinner = document.createElement("span");

  checkBtn.prepend(spinner);
  labelEl.textContent = "Analyzing…";

  // Simulated async delay so it feels like work happened
  setTimeout(() => {
    const result = analyze(text);
    render(result);
    spinner.remove();
    iconEl.style.display = "";
    labelEl.textContent = originalLabel;
    checkBtn.disabled = false;
  }, 650 + Math.random() * 400);
});
